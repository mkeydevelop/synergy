file(GLOB headers "*.h")
file(GLOB sources "*.cpp")

include_directories(
	../
	../../lib/
)

if (UNIX)
	include_directories(
		../../..
	)
elseif (WIN32)
	if (CASE_SENSITIVE)
		include_directories(
			../../lib/arch/mingw
		)
	endif()
endif()

if (WIN32)
  add_executable(synergyd WIN32 ${sources})
else()
  add_executable(synergyd ${sources})
endif()

if (NOT MINGW)
	target_link_libraries(synergyd
		arch base common io ipc mt net platform synergy ${libs})
else()
	target_link_libraries(synergyd
		-static -Wl,--start-group arch base common io ipc mt net platform server client synergy -Wl,--end-group ${libs})
endif()

if (CONF_CPACK)
	install(TARGETS
		synergyd
		COMPONENT core
		DESTINATION bin)
endif()
